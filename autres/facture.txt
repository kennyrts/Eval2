SELECT 
    pe.name as payment_entry_id,
    pe.posting_date as payment_date,
    pe.paid_amount,
    pe.payment_type,
    pe.mode_of_payment,
    pe.party as supplier,
    pe.reference_no,
    per.reference_name as purchase_invoice_id,
    per.allocated_amount,
    pi.grand_total as invoice_total,
    pi.purchase_order as purchase_order_id
FROM 
    `tabPayment Entry` pe
    INNER JOIN `tabPayment Entry Reference` per ON pe.name = per.parent
    INNER JOIN `tabPurchase Invoice` pi ON per.reference_name = pi.name
WHERE 
    pi.purchase_order = 'ACC-PINV-2025-00009'
    AND pe.docstatus = 1
ORDER BY 
    pe.posting_date DESC;
